import pathlib
load_widget = widgets.LoadWidget(wrapped_func=lambda filename: filename,
                                inputs=(inputs.FileInput('filename'), ),
                                button_name='Load',
                                layout='row wrap',
                                obj_name_generator=lambda kwargs: pathlib.Path(kwargs['filename']).stem,
                                hide_code=False)
